REIA Tracking Pixel - Installation Guide

This guide contains two pixel snippets to install on your website. Follow the steps below carefully.

Before you start: Replace REPLACE_WITH_EDUCATOR_ID in both snippets below with your unique educator ID provided by Utah REIA.

Snippet 1 - Landing Page Pixel

Paste this code into the HEAD section of your landing page. This is the page visitors land on when they click from the Utah REIA website.

This snippet captures UTM values when a REIA visitor arrives and stores them so they survive through your entire checkout flow.
Where to paste: Your landing page head section — in Kajabi, Teachable, or your CMS go to Page Settings → Tracking Codes → Head and paste it there.
<!-- REIA Landing Page Pixel -->
<script>
(function() {
  var EDUCATOR_ID = 'REPLACE_WITH_EDUCATOR_ID';
  var PIXEL_URL = 'https://utm-tracking-three.vercel.app/api/pixel';

  // Capture UTM values when visitor arrives from REIA
  var params = new URLSearchParams(window.location.search);
  var utmSource = params.get('utm_source');

  if (utmSource === 'utahreia') {
    sessionStorage.setItem('reia_utm_source',   utmSource);
    sessionStorage.setItem('reia_utm_medium',   params.get('utm_medium')   || '');
    sessionStorage.setItem('reia_utm_campaign', params.get('utm_campaign') || '');
    sessionStorage.setItem('reia_utm_content',  params.get('utm_content')  || '');
    sessionStorage.setItem('reia_gclid',        params.get('gclid')        || '');
    sessionStorage.setItem('reia_fbclid',       params.get('fbclid')       || '');
  }

  // Fire page_view event if visitor came from REIA
  var storedSource = sessionStorage.getItem('reia_utm_source');
  if (storedSource === 'utahreia') {
    fetch(PIXEL_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        event:        'page_view',
        educator_id:  EDUCATOR_ID,
        utm_source:   storedSource,
        utm_medium:   sessionStorage.getItem('reia_utm_medium'),
        utm_campaign: sessionStorage.getItem('reia_utm_campaign'),
        utm_content:  sessionStorage.getItem('reia_utm_content'),
        page_url:     window.location.href,
        referrer:     document.referrer
      })
    }).catch(function() {});
  }
})();
</script>

Snippet 2 - Checkout Success Page Pixel

Paste this code into the HEAD section of your checkout success or thank you page. This is the page shown after a purchase is completed.

This snippet fires a purchase event back to REIA so they can confirm the conversion and attribute the sale correctly.
Where to paste: Your order confirmation or thank you page head section only. Do NOT paste this on the landing page.
Optional: If your platform makes order data available as JavaScript variables, set these values before the snippet runs so REIA receives the full order details:

window.REIA_ORDER_VALUE = 497.00;
window.REIA_ORDER_ID = 'your_order_id';
window.REIA_PRODUCT_ID = 'your_product_id';
window.REIA_COUPON_CODE = 'coupon_used_or_empty'; 
<!-- REIA Purchase Pixel - SUCCESS PAGE ONLY -->
<script>
(function() {
  var EDUCATOR_ID = 'REPLACE_WITH_EDUCATOR_ID';
  var PIXEL_URL = 'https://utm-tracking-three.vercel.app/api/pixel';

  // Only fire if this visitor originally came from REIA
  var storedSource = sessionStorage.getItem('reia_utm_source');

  if (storedSource === 'utahreia') {
    fetch(PIXEL_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        event:        'purchase',
        educator_id:  EDUCATOR_ID,
        utm_source:   storedSource,
        utm_medium:   sessionStorage.getItem('reia_utm_medium'),
        utm_campaign: sessionStorage.getItem('reia_utm_campaign'),
        utm_content:  sessionStorage.getItem('reia_utm_content'),
        order_value:  window.REIA_ORDER_VALUE  || null,
        order_id:     window.REIA_ORDER_ID     || null,
        product_id:   window.REIA_PRODUCT_ID   || null,
        coupon_code:  window.REIA_COUPON_CODE  || null,
        page_url:     window.location.href
      })
    }).catch(function() {});
  }
})();
</script>
